# 1.Oracle 11.2.0.4 基于CentOS 7.6

（1）安装依赖包

```bash
yum update -y
yum install vim -y
yum install unzip -y

su - root

groupadd -g 54321 oinstall
groupadd -g 54322 dba
groupadd -g 54323 oper
groupadd -g 54327 asmdba
groupadd -g 54328 asmoper
groupadd -g 54329 asmadmin

useradd -u 54321 -g oinstall -G dba,oper,asmadmin oracle

passwd oracle
#上传Oracle安装文件
#上传p13390677_112040_Linux-x86-64_1of7.zip 到/home/oracle
#上传p13390677_112040_Linux-x86-64_2of7.zip 到/home/oracle
```

(2)设置linux内核参数

```bash
vim  /etc/sysctl.conf
#在conf文件最后添加以下几行
kernel.shmmni = 4096 
kernel.shmmax = 4398046511104
kernel.shmall = 1073741824
kernel.sem = 250 32000 100 128

fs.aio-max-nr = 1048576
fs.file-max = 6815744
net.ipv4.ip_local_port_range = 9000 65500
net.core.rmem_default = 262144
net.core.rmem_max = 4194304
net.core.wmem_default = 262144
net.core.wmem_max = 1048586
#加载修改的内核参数信息
/sbin/sysctl -p

#修改认证模块
vim /etc/security/limits.conf
#复制如下字段到limits.conf中
oracle   soft   nproc    131072
oracle   hard   nproc    131072
oracle   soft   nofile   131072
oracle   hard   nofile   131072
oracle   soft   core     unlimited
oracle   hard   core     unlimited
oracle   soft   memlock  50000000
oracle   hard   memlock  50000000

#修改/etc/hosts的主机与IP的映射
#如果之前没有设置过主机名，可以使用
#hostnamectl set-hostname za-db-center
vim /etc/hosts
#hosts文件最后一行添加
192.168.0.108 za-db-center
```

(3) 设置系统与用户环境变量

```bash
#设置系统环境变量
vim /etc/profile
#在profile文件最后添加
export TMP=/tmp

export ORACLE_HOSTNAME=za-db-center
export ORACLE_UNQNAME=zakss
export ORACLE_BASE=/ora01/app/oracle
export ORACLE_HOME=$ORACLE_BASE/product/11.2.0/db_1
export ORACLE_SID=zakss

PATH=/usr/sbin:$PATH:$ORACLE_HOME/bin

export LD_LIBRARY_PATH=$ORACLE_HOME/lib:/lib:/usr/lib;
export CLASSPATH=$ORACLE_HOME/jlib:$ORACLE_HOME/rdbms/jlib;

alias cdob='cd $ORACLE_BASE'
alias cdoh='cd $ORACLE_HOME'
alias tns='cd $ORACLE_HOME/network/admin'
alias envo='env | grep ORACLE'

umask 022

#环境变量的生效
source /etc/profile
```

(5) 关闭防火墙\添加权限

```bash
systemctl stop firewalld
systemctl disable firewalld

#增加权限
mkdir -p /ora01/app/oracle/product/11.2.0/db_1
chown oracle:oinstall -R /ora01

#SELINUX降级
SELINUX=permissive
setenforce Permissive
```

(6) 切换Oracle用户

```bash
su - oracle
cd /home/oracle
unzip p13390677_112040_Linux-x86-64_1of7.zip
unzip p13390677_112040_Linux-x86-64_2of7.zip

#检查命令别名
alias envo cdob cdoh tns
envo
cdob
cdoh
```

(7) 切换到Response目录下

```bash
cd /home/oracle/database/response
#备份文件
cp db_install.rsp db_install.rsp.bck
#修改文件，按照2.db_install.rsp进行修改
vim db_install.rsp
```

(8) 静默安装Oracle

```bash
cd /home/oracle/database

./runInstaller -silent \
-responseFile /home/oracle/database/response/db_install.rsp

#As a root user, execute the following script(s):
#    1. /ora01/app/oraInventory/orainstRoot.sh
#    2. /ora01/app/oracle/product/11.2.0/db_1/root.sh

#Successfully Setup Software.
su - root
/ora01/app/oraInventory/orainstRoot.sh
/ora01/app/oracle/product/11.2.0/db_1/root.sh
```

(9) 启动监听进程

```bash
su - oracle
#测试sqlplus
sqlplus / as sysdba
cd /home/oracle/database/response
cp netca.rsp netca.rsp.bck
#执行启动监听
netca -silent -responseFile /home/oracle/database/response/netca.rsp
```

(10) 新建数据库

```bash
cd /home/oracle/database/response
cp dbca.rsp dbca.rsp.bck
vim dbca.rsp

#修改新建数据库的信息，详见dbca.rsp
cd /home/oracle/database
dbca -silent -responseFile /home/oracle/database/response/dbca.rsp
```

(11) 无法启动，需要复制pfile参数文件到指定目录

```bash
#一定要使用oracle用户对文件操作
su - oracle
cd /ora01/app/oracle/admin/zakss/pfile
cp init.ora.812201910442 /ora01/app/oracle/product/11.2.0/db_1/dbs/initzakss.ora

sqlplus / as sysdba
startup
#如果提示ORA-01102: cannot mount database in EXCLUSIVE mode,则需要重启操作系统
```
