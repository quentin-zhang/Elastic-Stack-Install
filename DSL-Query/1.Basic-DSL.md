# 1.基础DSL

（1）集群与节点设置

```text
//集群简况状况
GET /_cluster/health
//获取集群设置
GET _cluster/settings
//集群节点情况
GET /_cat/allocation?v
//获取所有节点情况
GET /_cat/nodes?v
//获取所有索引
GET /_cat/indices?v
//查看以filebeat-开头命名的索引
GET /_cat/indices/filebeat-*?v&s=index
//查看分片概况
GET /_cat/shards
//查看分片情况以及未分配原因
GET _cat/shards?h=index,shard,prirep,state,unassigned.reason
//查看未分配分片的详细原因
GET /_cluster/allocation/explain?pretty
//获取索引的设置
GET /index_name/_settings
```

```json
//禁止通配符删除索引
PUT /_cluster/settings
{
    "persistent": {
        "action.destructive_requires_name": true
    }
}
//
//节点磁盘使用量超过85%将不分配新分片
//节点磁盘可用量最低为30gb，低于30gb则不分配分片
//每分钟更新磁盘信息
PUT _cluster/settings
{
    "transient": {
        "cluster.routing.allocation.disk.watermark.low": "85%",
        "cluster.routing.allocation.disk.watermark.high": "30gb",
        "cluster.info.update.interval": "1m"
    }
}
//
//设置集群的索引分片数量，1个主分片，1个副本分片
PUT /_template/index_defaults 
{
  "template": "*",
  "settings": {
    "number_of_shards": 1,
    "number_of_replicas" : 1
  }
}
//延迟副本重新分配
PUT index_name/_settings
{
    "settings": {
        "index.unassigned.node_left.delayed_timeout": "5m"
    }
}
```

（3）索引操作

```text
   //关闭索引
   POST /index_name/_close
   //打开索引
   POST /idnex_name/_open

  
```

```json
//重新构建索引
POST /_reindex
{
  "source": {
    "index": "source_index"
  }
  , "dest": {
    "index": "target_index",
    "version_type": "external"
  }
}

//索引备份（重置主分片数和副本分片数）
//步骤1
PUT /source_index/_settings
{
  "settings": {
    "index.routing.allocation.require._name": "nodename",
    "index.blocks.write": true
  }
}
//步骤2
POST /source_index/_shrink/target_index
{
    "settings": {
    "index.number_of_replicas": 1
    "index.number_of_shards": 1,
    "index.codec": "best_compression"
  }
}

//修改字段属性
PUT index_name/_mapping/doc
{
  "properties": {
    "body_bytes_sent": { 
      "type":     "number",
      "fielddata": true
    }
  }
}
```
